 \begin{enumerate}

  \item 14M.2.sl.TZ2.2\\
  Let $f(x)=5-x^2$. Part of the graph of $f$ is shown in the following diagram.
    \begin{center}
      \begin{tikzpicture}[xscale=0.75, yscale=0.75]
        \draw [thick, ->] (-4,0) -- (4,0) node [below] {$x$};
        \draw [thick, ->] (0,-3) -- (0,6) node [left] {$y$};
        \draw[thick, domain=-2.8:2.8] plot[samples=100](\x, {5-(\x)^2});
        \fill (2.236, 0) circle(3pt) node[above right]{$B$};
        \fill (-2.236, 0) circle(3pt) node[above left]{$A$};
      \end{tikzpicture}
    \end{center}
    The graph crosses the $x$-axis at the points $A$ and $B$.
    \begin{enumerate}
      \item Find the $x$-coordinate of $A$ and $B$. [3 marks]
      \item The region enclosed by the graph of $f$ and the $x$-axis is rotated $360^\circ$ about the $x$-axis.\\
      Find the volume of the solid formed. [3 marks]
    \end{enumerate}

\end{enumerate}
